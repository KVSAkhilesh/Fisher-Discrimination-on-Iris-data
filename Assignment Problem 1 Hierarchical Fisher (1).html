def fisher_proj(xtrain, class1, class2):
    
    X = xtrain[['Sepal.Length','Sepal.Width','Petal.Length','Petal.Width']].values
    y = xtrain['Species'].values

    label_dict = {1: class1, 2: class2}
    enc = LabelEncoder()
    labels = enc.fit(y)
    y = labels.transform(y) + 1

    meanvect = []
    for cl in range(1,3):
        meanvect.append(np.mean(xtrain[y==cl], axis=0))

    SW = np.zeros((4,4))
    meanvect = np.matrix(meanvect)
    for cl,mv in zip(range(1,4), meanvect):
        temp1 = np.zeros((4,4))
        for r in X[y == cl]:
            r, mv = r.reshape(4,1), mv.reshape(4,1)
            temp1 += (r-mv).dot((r-mv).T)
        SW += temp1

    overallmean = np.mean(X, axis=0)

    SB = np.zeros((4,4))
    for i,mv in enumerate(meanvect):
        n = X[y==i+1,:].shape[0]
        mv = mv.reshape(4,1)
        overallmean = overallmean.reshape(4,1)
        SB += n * (mv - overallmean).dot((mv - overallmean).T)
    
    eigvals, eigvec = np.linalg.eig(np.linalg.inv(SW).dot(SB))
    eigpairs = [(np.abs(eigvals[i]), eigvec[:,i]) for i in range(len(eigvals))]
    eigpairs = sorted(eigpairs, key=lambda k: k[0], reverse=True)
    print(eigpairs)

    weight = np.hstack((eigpairs[0][1].reshape(4,1), eigpairs[1][1].reshape(4,1)))
    print(weight.real)
    return weight.real
